CREATE DEFINER=`root`@`localhost` PROCEDURE `reset_event_complete`(IN in_event_string varchar(50))
BEGIN

case

when in_event_string = "individual_patterns" then update registration_data_raw set individual_patterns_complete = 0;
when in_event_string = "individual_sparring" then update registration_data_raw set individual_sparring_complete = 0;
when in_event_string = "individual_special_technique" then update registration_data_raw set individual_special_technique_complete = 0;
when in_event_string = "individual_power_test" then update registration_data_raw set individual_power_test_complete = 0;
when in_event_string = "team_prearranged_sparring" then update registration_data_raw set team_prearranged_sparring_complete = 0;
when in_event_string = "team_patterns" then update registration_data_raw set team_patterns_complete = 0;
when in_event_string = "team_sparring" then update registration_data_raw set team_sparring_complete = 0;
when in_event_string = "team_special_technique" then update registration_data_raw set team_special_technique_complete = 0;
when in_event_string = "team_power_test" then update registration_data_raw set team_power_test_complete = 0;

end case

;
END
