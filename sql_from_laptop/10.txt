CREATE 
    ALGORITHM = UNDEFINED 
    DEFINER = `root`@`localhost` 
    SQL SECURITY DEFINER
VIEW `incomplete_entries` AS
    SELECT 
        `registration_data_raw`.`id` AS `id`,
        `registration_data_raw`.`first_name` AS `first_name`,
        `registration_data_raw`.`last_name` AS `last_name`,
        `registration_data_raw`.`date_of_birth` AS `date_of_birth`,
        `registration_data_raw`.`gender` AS `gender`,
        `registration_data_raw`.`black_belt` AS `black_belt`,
        `registration_data_raw`.`gup_dan` AS `gup_dan`,
        `registration_data_raw`.`height` AS `height`,
        `registration_data_raw`.`weight` AS `weight`,
        `registration_data_raw`.`team` AS `team`,
        `registration_data_raw`.`individual_patterns` AS `individual_patterns`,
        `registration_data_raw`.`individual_patterns_complete` AS `individual_patterns_complete`,
        `registration_data_raw`.`individual_sparring` AS `individual_sparring`,
        `registration_data_raw`.`individual_sparring_complete` AS `individual_sparring_complete`,
        `registration_data_raw`.`individual_special_technique` AS `individual_special_technique`,
        `registration_data_raw`.`individual_special_technique_complete` AS `individual_special_technique_complete`,
        `registration_data_raw`.`individual_power_test` AS `individual_power_test`,
        `registration_data_raw`.`individual_power_test_complete` AS `individual_power_test_complete`,
        `registration_data_raw`.`team_prearranged_sparring` AS `team_prearranged_sparring`,
        `registration_data_raw`.`team_prearranged_sparring_complete` AS `team_prearranged_sparring_complete`,
        `registration_data_raw`.`team_patterns` AS `team_patterns`,
        `registration_data_raw`.`team_patterns_complete` AS `team_patterns_complete`,
        `registration_data_raw`.`team_sparring` AS `team_sparring`,
        `registration_data_raw`.`team_sparring_complete` AS `team_sparring_complete`,
        `registration_data_raw`.`team_special_technique` AS `team_special_technique`,
        `registration_data_raw`.`team_special_technique_complete` AS `team_special_technique_complete`,
        `registration_data_raw`.`team_power_test` AS `team_power_test`,
        `registration_data_raw`.`team_power_test_complete` AS `team_power_test_complete`
    FROM
        `registration_data_raw`
    WHERE
        ((`registration_data_raw`.`first_name` = 'empty')
            OR (`registration_data_raw`.`last_name` = 'empty')
            OR (`registration_data_raw`.`date_of_birth` = '2099-01-01')
            OR (`registration_data_raw`.`gender` = 'empty')
            OR (`registration_data_raw`.`height` = 0)
            OR (`registration_data_raw`.`weight` = 0)
            OR (`registration_data_raw`.`team` = 'empty'))
